using namespace std;

int T, n, m, k, answer[2001];
int visited[501], recStack[501];

int checkCycle(int x)
{
    if (!visited[x])
    {
        visited[x] = true;
        recStack[x] = true;
        for (int i = 0; i < edges[x].size(); i++)
        {
            int next = edges[x][i];
            if (!visited[next] && checkCycle(next))
                return true;

            else if (recStack[next])

    recStack[x] = false;
    return false;

int run(int index)

    if (index == k)

        edges[x].pop_back();
        edges[y].push_back(x);
    }

    return false;

}
